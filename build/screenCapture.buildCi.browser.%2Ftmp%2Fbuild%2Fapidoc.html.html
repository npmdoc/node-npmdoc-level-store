function level-store.indexes.chunks (db, key)

  • description and source-code
    chunks = function (db, key) {
  var idx = {};
  var chunks = 0;

  idx.newKey = function () {
    return key + ' ' + padHex(chunks++);
  };

  idx.modKey = padHex;

  idx.initialize = function (cb) {
    peek.last(db, {
      reverse : true,
      start : key + ' ',
      end : key + '~'
    }, function (err, lastKey) {
      if (!err) chunks = unpadHex(lastKey.substr(key.length + 1));
      cb(null);
    });
  }

  idx.parseIndex = function (chunk) {
    return {
      index: unpadHex(chunk.index),
      data: chunk.data
    };
  }

  return idx;
}
